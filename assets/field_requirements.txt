# FireCheck — Field Requirements Config (Issue #43)
#
# Toggle whether each form field is required or optional WITHOUT rebuilding
# the app. Two ways to update:
#   1. Drop a file named `field_requirements.txt` in the same Google Drive
#      (or FTP) folder that holds the shapefile components. It downloads
#      with the assignment and overrides this bundled copy.
#   2. Edit this bundled asset and rebuild — used when no sidecar file
#      has been imported yet.
#
# Format: <field_key> = required | optional
# Lines starting with `#` and blank lines are ignored.
# Unknown keys are ignored. Unknown values default to `required`.
#
# Fallback: if this file is missing or unparseable, ALL fields are treated
# as required (parity with the pre-Issue-#43 behavior).

# Building form
building.buildingName        = required
building.ra9514Type          = required
building.storeys             = required
building.material            = required
building.cost                = required
building.fireLoad            = required
building.photo               = required

# Road form
road.widthMeters             = required
road.roadFeatures            = optional
road.othersDescription       = required
road.photo                   = required
